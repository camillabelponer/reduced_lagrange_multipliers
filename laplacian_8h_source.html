117template <int dim, int spacedim>
118ProblemParameters<dim, spacedim>::ProblemParameters()
119 : ParameterAcceptor("/Immersed Problem/")
120 , rhs("/Immersed Problem/Right hand side")
121 , bc("/Immersed Problem/Dirichlet boundary conditions")
122 , inner_control("/Immersed Problem/Solver/Inner control")
123 , outer_control("/Immersed Problem/Solver/Outer control")
124{
125 add_parameter("FE degree", fe_degree, "", this->prm, Patterns::Integer(1));
126 add_parameter("Output directory", output_directory);
127 add_parameter("Output name", output_name);
128 add_parameter("Output results also before solving",
129 output_results_before_solving);
130 add_parameter("Initial refinement", initial_refinement);
131 add_parameter("Dirichlet boundary ids", dirichlet_ids);
132 enter_subsection("Grid generation");
133 {
134 add_parameter("Grid generator", name_of_grid);
135 add_parameter("Grid generator arguments", arguments_for_grid);
136 }
137 leave_subsection();
138 enter_subsection("Refinement and remeshing");
139 {
140 add_parameter("Strategy",
141 refinement_strategy,
142 "",
143 this->prm,
144 Patterns::Selection("fixed_fraction|fixed_number|global"));
145 add_parameter("Coarsening fraction", coarsening_fraction);
146 add_parameter("Refinement fraction", refinement_fraction);
147 add_parameter("Maximum number of cells", max_cells);
148 add_parameter("Number of refinement cycles", n_refinement_cycles);
149 }
150 leave_subsection();
151
152 this->prm.enter_subsection("Error");
153 convergence_table.add_parameters(this->prm);
154 this->prm.leave_subsection();
155}
