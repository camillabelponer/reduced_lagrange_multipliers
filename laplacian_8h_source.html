260template <int dim, int spacedim>
261void
262read_grid_and_cad_files(const std::string &grid_file_name,
263 const std::string &ids_and_cad_file_names,
264 Triangulation<dim, spacedim> &tria)
265{
266 GridIn<dim, spacedim> grid_in;
267 grid_in.attach_triangulation(tria);
268 grid_in.read(grid_file_name);
269#ifdef DEAL_II_WITH_OPENCASCADE
270 using map_type = std::map<types::manifold_id, std::string>;
271 using Converter = Patterns::Tools::Convert<map_type>;
272 for (const auto &pair : Converter::to_value(ids_and_cad_file_names))
273 {
274 const auto &manifold_id = pair.first;
275 const auto &cad_file_name = pair.second;
276 const auto extension = boost::algorithm::to_lower_copy(
277 cad_file_name.substr(cad_file_name.find_last_of('.') + 1));
278 TopoDS_Shape shape;
279 if (extension == "iges" || extension == "igs")
280 shape = OpenCASCADE::read_IGES(cad_file_name);
281 else if (extension == "step" || extension == "stp")
282 shape = OpenCASCADE::read_STEP(cad_file_name);
283 else
284 AssertThrow(false,
285 ExcNotImplemented("We found an extension that we "
286 "do not recognize as a CAD file "
287 "extension. Bailing out."));
288 const auto n_elements = OpenCASCADE::count_elements(shape);
289 if ((std::get<0>(n_elements) == 0))
290 tria.set_manifold(
291 manifold_id,
292 OpenCASCADE::ArclengthProjectionLineManifold<dim, spacedim>(shape));
293 else if (spacedim == 3)
294 {
295 const auto t = reinterpret_cast<Triangulation<dim, 3> *>(&tria);
296 t->set_manifold(manifold_id,
297 OpenCASCADE::NormalToMeshProjectionManifold<dim, 3>(
298 shape));
299 }
300 else
301 tria.set_manifold(manifold_id,
302 OpenCASCADE::NURBSPatchManifold<dim, spacedim>(
303 TopoDS::Face(shape)));
304 }
305#else
306 (void)ids_and_cad_file_names;
307 AssertThrow(false, ExcNotImplemented("Generation of the grid failed."));
308#endif
309}

334template <int dim, int spacedim>
335void
336PoissonProblem<dim, spacedim>::setup_inclusions_particles()
337{
338 inclusions_as_particles.clear();
339 inclusions_as_particles.initialize(tria, StaticMappingQ1<spacedim>::mapping);
340
341 if (par.inclusions.empty())
342 return;
343
344 std::vector<Point<spacedim>> particles_positions;
345 particles_positions.reserve(inclusion->n_q_points * par.inclusions.size());
346 for (unsigned int i = 0; i < par.inclusions.size(); ++i)
347 {
348 const auto &p = inclusion->get_current_support_points(par.inclusions[i]);
349 particles_positions.insert(particles_positions.end(), p.begin(), p.end());
350 }
351
352 std::vector<BoundingBox<spacedim>> all_boxes;
353 all_boxes.reserve(tria.n_locally_owned_active_cells());
354 for (const auto &cell : tria.active_cell_iterators())
355 if (cell->is_locally_owned())
356 all_boxes.emplace_back(cell->bounding_box());
357 const auto tree = pack_rtree(all_boxes);
358 const auto local_boxes =
359 extract_rtree_level(tree, par.rtree_extraction_level);
360
361 global_bounding_boxes =
362 Utilities::MPI::all_gather(mpi_communicator, local_boxes);
363
364 Assert(!global_bounding_boxes.empty(),
365 ExcInternalError(
366 "I was expecting the "
367 "global_bounding_boxes to be filled at this stage. "
368 "Make sure you fill this vector before trying to use it "
369 "here. Bailing out."));
370 inclusions_as_particles.insert_global_particles(particles_positions,
371 global_bounding_boxes);
372 tria.signals.pre_distributed_refinement.connect(
373 [&]() { inclusions_as_particles.register_store_callback_function(); });
374 tria.signals.post_distributed_refinement.connect(
375 [&]() { inclusions_as_particles.register_load_callback_function(false); });
376 pcout << "Inclusions particles: "
377 << inclusions_as_particles.n_global_particles() << std::endl;
378}

407template <int dim, int spacedim>
408void
409PoissonProblem<dim, spacedim>::setup_dofs()
410{
411 TimerOutput::Scope t(computing_timer, "Setup dofs");
412 dh.distribute_dofs(*fe);
413
414 owned_dofs.resize(2);
415 owned_dofs[0] = dh.locally_owned_dofs();
416 relevant_dofs.resize(2);
417 DoFTools::extract_locally_relevant_dofs(dh, relevant_dofs[0]);
418 {
419 constraints.reinit(relevant_dofs[0]);
420 DoFTools::make_hanging_node_constraints(dh, constraints);
421 for (const auto id : par.dirichlet_ids)
422 VectorTools::interpolate_boundary_values(dh, id, par.bc, constraints);
423 constraints.close();
424 }
425 {
426 stiffness_matrix.clear();
427 DynamicSparsityPattern dsp(relevant_dofs[0]);
428 DoFTools::make_sparsity_pattern(dh, dsp, constraints, false);
429 SparsityTools::distribute_sparsity_pattern(dsp,
430 owned_dofs[0],
431 mpi_communicator,
432 relevant_dofs[0]);
433 stiffness_matrix.reinit(owned_dofs[0],
434 owned_dofs[0],
435 dsp,
436 mpi_communicator);
437 }
438 {
439 auto inclusions_set =
440 Utilities::MPI::create_evenly_distributed_partitioning(
441 mpi_communicator, par.inclusions.size());
442
443 owned_dofs[1] = inclusions_set.tensor_product(
444 complete_index_set(par.n_fourier_coefficients));
445
446 coupling_matrix.clear();
447 DynamicSparsityPattern dsp(dh.n_dofs(), n_inclusions_dofs());
448
449 relevant_dofs[1] = assemble_coupling_sparsity(dsp);
450 SparsityTools::distribute_sparsity_pattern(dsp,
451 owned_dofs[0],
452 mpi_communicator,
453 relevant_dofs[0]);
454 coupling_matrix.reinit(owned_dofs[0], owned_dofs[1], dsp, mpi_communicator);
455 inclusion_constraints.close();
456 }
457
458 locally_relevant_solution.reinit(owned_dofs, relevant_dofs, mpi_communicator);
459 system_rhs.reinit(owned_dofs, mpi_communicator);
460 solution.reinit(owned_dofs, mpi_communicator);
461
462 pcout << " Number of degrees of freedom: " << owned_dofs[0].size() << " + "
463 << owned_dofs[1].size()
464 << " (locally owned: " << owned_dofs[0].n_elements() << " + "
465 << owned_dofs[1].n_elements() << ")" << std::endl;
466}

468template <int dim, int spacedim>
469void
470PoissonProblem<dim, spacedim>::assemble_poisson_system()
471{
472 stiffness_matrix = 0;
473 coupling_matrix = 0;
474 system_rhs = 0;
475 TimerOutput::Scope t(computing_timer, "Assemble Stiffness");
476 FEValues<spacedim> fe_values(*fe,
477 *quadrature,
478 update_values | update_gradients |
479 update_quadrature_points | update_JxW_values);
480 const unsigned int dofs_per_cell = fe->n_dofs_per_cell();
481 const unsigned int n_q_points = quadrature->size();
482 FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
483 Vector<double> cell_rhs(dofs_per_cell);
484 std::vector<double> rhs_values(n_q_points);
485 std::vector<Tensor<1, spacedim>> grad_phi_u(dofs_per_cell);
486 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
487 const FEValuesExtractors::Scalar scalar(0);
488 for (const auto &cell : dh.active_cell_iterators())
489 if (cell->is_locally_owned())
490 {
491 cell_matrix = 0;
492 cell_rhs = 0;
493 fe_values.reinit(cell);
494 par.rhs.value_list(fe_values.get_quadrature_points(), rhs_values);
495 for (unsigned int q = 0; q < n_q_points; ++q)
496 {
497 for (unsigned int k = 0; k < dofs_per_cell; ++k)
498 grad_phi_u[k] = fe_values[scalar].gradient(k, q);
499 for (unsigned int i = 0; i < dofs_per_cell; ++i)
500 {
501 for (unsigned int j = 0; j < dofs_per_cell; ++j)
502 {
503 cell_matrix(i, j) +=
504 grad_phi_u[i] * grad_phi_u[j] * fe_values.JxW(q);
505 }
506 cell_rhs(i) += fe_values.shape_value(i, q) * rhs_values[q] *
507 fe_values.JxW(q);
508 }
509 }
510 cell->get_dof_indices(local_dof_indices);
511 constraints.distribute_local_to_global(cell_matrix,
512 cell_rhs,
513 local_dof_indices,
514 stiffness_matrix,
515 system_rhs.block(0));
516 }
517 stiffness_matrix.compress(VectorOperation::add);
518 system_rhs.compress(VectorOperation::add);
519}

523template <int dim, int spacedim>
524IndexSet
525PoissonProblem<dim, spacedim>::assemble_coupling_sparsity(
526 DynamicSparsityPattern &dsp) const
527{
528 TimerOutput::Scope t(computing_timer, "Assemble Coupling sparsity");
529 IndexSet relevant(n_inclusions_dofs());
530
531 const FEValuesExtractors::Scalar scalar(0);
532
533 std::vector<types::global_dof_index> dof_indices(fe->n_dofs_per_cell());
534 std::vector<types::global_dof_index> inclusion_dof_indices;
535
536 auto particle = inclusions_as_particles.begin();
537 while (particle != inclusions_as_particles.end())
538 {
539 const auto &cell = particle->get_surrounding_cell(tria);
540 const auto &dh_cell =
541 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
542 dh_cell->get_dof_indices(dof_indices);
543
544 const auto pic = inclusions_as_particles.particles_in_cell(cell);
545 Assert(pic.begin() == particle, ExcInternalError());
546 std::set<types::global_dof_index> inclusion_dof_indices_set;
547 for (const auto &p : pic)
548 {
549 const auto ids = inclusion->get_dof_indices(p.get_id());
550 inclusion_dof_indices_set.insert(ids.begin(), ids.end());
551 }
552 inclusion_dof_indices.resize(0);
553 inclusion_dof_indices.insert(inclusion_dof_indices.begin(),
554 inclusion_dof_indices_set.begin(),
555 inclusion_dof_indices_set.end());
556
557 constraints.add_entries_local_to_global(dof_indices,
558 inclusion_dof_indices,
559 dsp);
560 relevant.add_indices(inclusion_dof_indices.begin(),
561 inclusion_dof_indices.end());
562 particle = pic.end();
563 }
564 return relevant;
565}

569template <int dim, int spacedim>
570void
571PoissonProblem<dim, spacedim>::assemble_coupling()
572{
573 TimerOutput::Scope t(computing_timer, "Assemble Coupling matrix");
574 const FEValuesExtractors::Scalar scalar(0);
575 std::vector<types::global_dof_index> fe_dof_indices(fe->n_dofs_per_cell());
576 std::vector<types::global_dof_index> inclusion_dof_indices(
577 inclusion->n_coefficients);
578
579 FullMatrix<double> local_matrix(fe->n_dofs_per_cell(),
580 inclusion->n_coefficients);
581 Vector<double> local_rhs(inclusion->n_coefficients);
582
583 auto particle = inclusions_as_particles.begin();
584 while (particle != inclusions_as_particles.end())
585 {
586 const auto &cell = particle->get_surrounding_cell(tria);
587 const auto &dh_cell =
588 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
589 dh_cell->get_dof_indices(fe_dof_indices);
590 const auto pic = inclusions_as_particles.particles_in_cell(cell);
591 Assert(pic.begin() == particle, ExcInternalError());
592 for (const auto &p : pic)
593 {
594 local_matrix = 0;
595 local_rhs = 0;
596 inclusion_dof_indices = inclusion->get_dof_indices(p.get_id());
597 const auto &inclusion_fe_values =
598 inclusion->reinit(p.get_id(), par.inclusions);
599
600 const auto &ref_q = p.get_reference_location();
601 const auto &real_q = p.get_location();
602 for (unsigned int j = 0; j < inclusion->n_coefficients; ++j)
603 {
604 for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
605 local_matrix(i, j) +=
606 fe->shape_value(i, ref_q) * inclusion_fe_values[j];
607 local_rhs(j) +=
608 inclusion_fe_values[j] * par.inclusions_rhs.value(real_q);
609 }
610 constraints.distribute_local_to_global(local_matrix,
611 fe_dof_indices,
612 inclusion_constraints,
613 inclusion_dof_indices,
614 coupling_matrix);
615 inclusion_constraints.distribute_local_to_global(
616 local_rhs, inclusion_dof_indices, system_rhs.block(1));
617 }
618 particle = pic.end();
619 }
620 coupling_matrix.compress(VectorOperation::add);
621 system_rhs.compress(VectorOperation::add);
622}

626template <int dim, int spacedim>
627void
628PoissonProblem<dim, spacedim>::solve()
629{
630 TimerOutput::Scope t(computing_timer, "Solve");
631 LA::MPI::PreconditionAMG prec_A;
632 {
633 LA::MPI::PreconditionAMG::AdditionalData data;
634#ifdef USE_PETSC_LA
635 data.symmetric_operator = true;
636#endif
637 prec_A.initialize(stiffness_matrix, data);
638 }
639
640 const auto A = linear_operator<LA::MPI::Vector>(stiffness_matrix);
641 const auto amgA = linear_operator(A, prec_A);
642
643 // LA::SolverCG cg_stiffness(par.inner_control);
644 SolverCG<LA::MPI::Vector> cg_stiffness(par.inner_control);
645 const auto invA = inverse_operator(A, cg_stiffness, amgA);
646
647 // Some aliases
648 auto &u = solution.block(0);
649 auto &lambda = solution.block(1);
650
651 const auto &f = system_rhs.block(0);
652 const auto &g = system_rhs.block(1);
653
654 if (par.inclusions.empty())
655 {
656 u = invA * f;
657 }
658 else
659 {
660 const auto Bt = linear_operator<LA::MPI::Vector>(coupling_matrix);
661 const auto B = transpose_operator(Bt);
662
663
664 // Schur complement
665 const auto S = B * invA * Bt;
666
667 LA::SolverCG cg_schur(par.outer_control);
668 // LA::SolverGMRES gmres_schur(solver_control);
669 SolverGMRES<LA::MPI::Vector> gmres_schur(par.outer_control);
670 const auto invS = inverse_operator(S, gmres_schur);
671
672
673 pcout << " f norm: " << f.l2_norm() << ", g norm: " << g.l2_norm()
674 << std::endl;
675
676 // Compute Lambda first
677 lambda = invS * (B * invA * f - g);
678 pcout << " Solved for lambda in " << par.outer_control.last_step()
679 << " iterations." << std::endl;
680
681 // Then compute u
682 u = invA * (f - Bt * lambda);
683 }
684
685 pcout << " Solved for u in " << par.inner_control.last_step()
686 << " iterations." << std::endl;
687 constraints.distribute(u);
688 inclusion_constraints.distribute(lambda);
689 locally_relevant_solution = solution;
690}
