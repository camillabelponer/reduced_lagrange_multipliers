44template <int dim, int spacedim>
45void
46read_grid_and_cad_files(const std::string & grid_file_name,
47 const std::string & ids_and_cad_file_names,
48 Triangulation<dim, spacedim> &tria)
49{
50 GridIn<dim, spacedim> grid_in;
51 grid_in.attach_triangulation(tria);
52 grid_in.read(grid_file_name);
53#ifdef DEAL_II_WITH_OPENCASCADE
54 using map_type = std::map<types::manifold_id, std::string>;
55 using Converter = Patterns::Tools::Convert<map_type>;
56 for (const auto &pair : Converter::to_value(ids_and_cad_file_names))
57 {
58 const auto &manifold_id = pair.first;
59 const auto &cad_file_name = pair.second;
60 const auto extension = boost::algorithm::to_lower_copy(
61 cad_file_name.substr(cad_file_name.find_last_of('.') + 1));
62 TopoDS_Shape shape;
63 if (extension == "iges" || extension == "igs")
64 shape = OpenCASCADE::read_IGES(cad_file_name);
65 else if (extension == "step" || extension == "stp")
66 shape = OpenCASCADE::read_STEP(cad_file_name);
67 else
68 AssertThrow(false,
69 ExcNotImplemented("We found an extension that we "
70 "do not recognize as a CAD file "
71 "extension. Bailing out."));
72 const auto n_elements = OpenCASCADE::count_elements(shape);
73 if ((std::get<0>(n_elements) == 0))
74 tria.set_manifold(
75 manifold_id,
76 OpenCASCADE::ArclengthProjectionLineManifold<dim, spacedim>(shape));
77 else if (spacedim == 3)
78 {
79 const auto t = reinterpret_cast<Triangulation<dim, 3> *>(&tria);
80 t->set_manifold(manifold_id,
81 OpenCASCADE::NormalToMeshProjectionManifold<dim, 3>(
82 shape));
83 }
84 else
85 tria.set_manifold(manifold_id,
86 OpenCASCADE::NURBSPatchManifold<dim, spacedim>(
87 TopoDS::Face(shape)));
88 }
89#else
90 (void)ids_and_cad_file_names;
91 AssertThrow(false, ExcNotImplemented("Generation of the grid failed."));
92#endif
93}

97template <int dim, int spacedim>
98void
99ElasticityProblem<dim, spacedim>::make_grid()
100{
101 try
102 {
103 GridGenerator::generate_from_name_and_arguments(tria,
104 par.name_of_grid,
105 par.arguments_for_grid);
106 }
107 catch (...)
108 {
109 pcout << "Generating from name and argument failed." << std::endl
110 << "Trying to read from file name." << std::endl;
111 read_grid_and_cad_files(par.name_of_grid, par.arguments_for_grid, tria);
112 }
113 tria.refine_global(par.initial_refinement);
114}

129template <int dim, int spacedim>
130void
131ElasticityProblem<dim, spacedim>::setup_dofs()
132{
133 TimerOutput::Scope t(computing_timer, "Setup dofs");
134 dh.distribute_dofs(*fe);
135
136 owned_dofs.resize(2);
137 owned_dofs[0] = dh.locally_owned_dofs();
138 relevant_dofs.resize(2);
139 DoFTools::extract_locally_relevant_dofs(dh, relevant_dofs[0]);
140 {
141 constraints.reinit(relevant_dofs[0]);
142 DoFTools::make_hanging_node_constraints(dh, constraints);
143 for (const auto id : par.dirichlet_ids)
144 VectorTools::interpolate_boundary_values(dh, id, par.bc, constraints);
145 constraints.close();
146 }
147 {
148 stiffness_matrix.clear();
149 DynamicSparsityPattern dsp(relevant_dofs[0]);
150 DoFTools::make_sparsity_pattern(dh, dsp, constraints, false);
151 SparsityTools::distribute_sparsity_pattern(dsp,
152 owned_dofs[0],
153 mpi_communicator,
154 relevant_dofs[0]);
155 stiffness_matrix.reinit(owned_dofs[0],
156 owned_dofs[0],
157 dsp,
158 mpi_communicator);
159 }
160 inclusion_constraints.close();
161
162 if (inclusions.n_dofs() > 0)
163 {
164 auto inclusions_set =
165 Utilities::MPI::create_evenly_distributed_partitioning(
166 mpi_communicator, inclusions.n_inclusions());
167
168 owned_dofs[1] = inclusions_set.tensor_product(
169 complete_index_set(inclusions.n_dofs_per_inclusion()));
170
171 coupling_matrix.clear();
172 DynamicSparsityPattern dsp(dh.n_dofs(), inclusions.n_dofs());
173
174 relevant_dofs[1] = assemble_coupling_sparsity(dsp);
175 SparsityTools::distribute_sparsity_pattern(dsp,
176 owned_dofs[0],
177 mpi_communicator,
178 relevant_dofs[0]);
179 coupling_matrix.reinit(owned_dofs[0],
180 owned_dofs[1],
181 dsp,
182 mpi_communicator);
183
184 DynamicSparsityPattern idsp(inclusions.n_dofs(), inclusions.n_dofs());
185 for (const auto i : owned_dofs[1])
186 idsp.add(i, i);
187
188 SparsityTools::distribute_sparsity_pattern(idsp,
189 owned_dofs[1],
190 mpi_communicator,
191 relevant_dofs[1]);
192 inclusion_matrix.reinit(owned_dofs[1],
193 owned_dofs[1],
194 idsp,
195 mpi_communicator);
196 }
197
198 locally_relevant_solution.reinit(owned_dofs, relevant_dofs, mpi_communicator);
199 system_rhs.reinit(owned_dofs, mpi_communicator);
200 solution.reinit(owned_dofs, mpi_communicator);
201
202 pcout << " Number of degrees of freedom: " << owned_dofs[0].size() << " + "
203 << owned_dofs[1].size()
204 << " (locally owned: " << owned_dofs[0].n_elements() << " + "
205 << owned_dofs[1].n_elements() << ")" << std::endl;
206}

208template <int dim, int spacedim>
209void
210ElasticityProblem<dim, spacedim>::assemble_elasticity_system()
211{
212 stiffness_matrix = 0;
213 coupling_matrix = 0;
214 system_rhs = 0;
215 TimerOutput::Scope t(computing_timer, "Assemble Stiffness");
216 FEValues<spacedim> fe_values(*fe,
217 *quadrature,
218 update_values | update_gradients |
219 update_quadrature_points | update_JxW_values);
220 const unsigned int dofs_per_cell = fe->n_dofs_per_cell();
221 const unsigned int n_q_points = quadrature->size();
222 FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
223 Vector<double> cell_rhs(dofs_per_cell);
224 std::vector<Vector<double>> rhs_values(n_q_points, Vector<double>(spacedim));
225 std::vector<Tensor<2, spacedim>> grad_phi_u(dofs_per_cell);
226 std::vector<double> div_phi_u(dofs_per_cell);
227 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
228
229 for (const auto &cell : dh.active_cell_iterators())
230 if (cell->is_locally_owned())
231 {
232 cell_matrix = 0;
233 cell_rhs = 0;
234 fe_values.reinit(cell);
235 par.rhs.vector_value_list(fe_values.get_quadrature_points(),
236 rhs_values);
237 for (unsigned int q = 0; q < n_q_points; ++q)
238 {
239 for (unsigned int k = 0; k < dofs_per_cell; ++k)
240 {
241 grad_phi_u[k] = fe_values[displacement].gradient(k, q);
242 div_phi_u[k] = fe_values[displacement].divergence(k, q);
243 }
244 for (unsigned int i = 0; i < dofs_per_cell; ++i)
245 {
246 for (unsigned int j = 0; j < dofs_per_cell; ++j)
247 {
248 cell_matrix(i, j) +=
249 (par.mu * scalar_product(grad_phi_u[i], grad_phi_u[j]) +
250 par.lambda * div_phi_u[i] * div_phi_u[j]) *
251 fe_values.JxW(q);
252 }
253 const auto comp_i = fe->system_to_component_index(i).first;
254 cell_rhs(i) += fe_values.shape_value(i, q) *
255 rhs_values[q][comp_i] * fe_values.JxW(q);
256 }
257 }
258 cell->get_dof_indices(local_dof_indices);
259 constraints.distribute_local_to_global(cell_matrix,
260 cell_rhs,
261 local_dof_indices,
262 stiffness_matrix,
263 system_rhs.block(0));
264 }
265 stiffness_matrix.compress(VectorOperation::add);
266 system_rhs.compress(VectorOperation::add);
267}

271template <int dim, int spacedim>
272IndexSet
273ElasticityProblem<dim, spacedim>::assemble_coupling_sparsity(
274 DynamicSparsityPattern &dsp) const
275{
276 TimerOutput::Scope t(computing_timer, "Assemble Coupling sparsity");
277 IndexSet relevant(inclusions.n_dofs());
278
279 std::vector<types::global_dof_index> dof_indices(fe->n_dofs_per_cell());
280 std::vector<types::global_dof_index> inclusion_dof_indices;
281
282 auto particle = inclusions.inclusions_as_particles.begin();
283 while (particle != inclusions.inclusions_as_particles.end())
284 {
285 const auto &cell = particle->get_surrounding_cell();
286 const auto dh_cell =
287 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
288 dh_cell->get_dof_indices(dof_indices);
289
290 const auto pic =
291 inclusions.inclusions_as_particles.particles_in_cell(cell);
292 Assert(pic.begin() == particle, ExcInternalError());
293 std::set<types::global_dof_index> inclusion_dof_indices_set;
294 for (const auto &p : pic)
295 {
296 const auto ids = inclusions.get_dof_indices(p.get_id());
297 inclusion_dof_indices_set.insert(ids.begin(), ids.end());
298 }
299 inclusion_dof_indices.resize(0);
300 inclusion_dof_indices.insert(inclusion_dof_indices.begin(),
301 inclusion_dof_indices_set.begin(),
302 inclusion_dof_indices_set.end());
303
304 constraints.add_entries_local_to_global(dof_indices,
305 inclusion_dof_indices,
306 dsp);
307 relevant.add_indices(inclusion_dof_indices.begin(),
308 inclusion_dof_indices.end());
309 particle = pic.end();
310 }
311 return relevant;
312}

316template <int dim, int spacedim>
317void
318ElasticityProblem<dim, spacedim>::assemble_coupling()
319{
320 TimerOutput::Scope t(computing_timer, "Assemble Coupling matrix");
321
322 const FEValuesExtractors::Scalar scalar(0);
323 std::vector<types::global_dof_index> fe_dof_indices(fe->n_dofs_per_cell());
324 std::vector<types::global_dof_index> inclusion_dof_indices(
325 inclusions.n_dofs_per_inclusion());
326
327 FullMatrix<double> local_coupling_matrix(fe->n_dofs_per_cell(),
328 inclusions.n_dofs_per_inclusion());
329
330 FullMatrix<double> local_bulk_matrix(fe->n_dofs_per_cell(),
331 fe->n_dofs_per_cell());
332
333 FullMatrix<double> local_inclusion_matrix(inclusions.n_dofs_per_inclusion(),
334 inclusions.n_dofs_per_inclusion());
335
336 Vector<double> local_rhs(inclusions.n_dofs_per_inclusion());
337
338 auto particle = inclusions.inclusions_as_particles.begin();
339 while (particle != inclusions.inclusions_as_particles.end())
340 {
341 const auto &cell = particle->get_surrounding_cell();
342 const auto dh_cell =
343 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
344 dh_cell->get_dof_indices(fe_dof_indices);
345 const auto pic =
346 inclusions.inclusions_as_particles.particles_in_cell(cell);
347 Assert(pic.begin() == particle, ExcInternalError());
348
349 auto p = pic.begin();
350 auto next_p = pic.begin();
351 while (p != pic.end())
352 {
353 const auto inclusion_id = inclusions.get_inclusion_id(p->get_id());
354 inclusion_dof_indices = inclusions.get_dof_indices(p->get_id());
355 local_coupling_matrix = 0;
356 local_inclusion_matrix = 0;
357 local_bulk_matrix = 0;
358 local_rhs = 0;
359
360 // Extract all points that refer to the same inclusion
361 std::vector<Point<spacedim>> ref_q_points;
362 for (; next_p != pic.end() &&
363 inclusions.get_inclusion_id(next_p->get_id()) == inclusion_id;
364 ++next_p)
365 ref_q_points.push_back(next_p->get_reference_location());
366 FEValues<spacedim, spacedim> fev(*fe,
367 ref_q_points,
368 update_values | update_gradients);
369 fev.reinit(dh_cell);
370 for (unsigned int q = 0; q < ref_q_points.size(); ++q)
371 {
372 const auto id = p->get_id();
373 const auto &inclusion_fe_values = inclusions.get_fe_values(id);
374 const auto &real_q = p->get_location();
375
376 // Coupling and inclusions matrix
377 for (unsigned int j = 0; j < inclusions.n_dofs_per_inclusion();
378 ++j)
379 {
380 for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
381 {
382 const auto comp_i =
383 fe->system_to_component_index(i).first;
384 if (comp_i == inclusions.get_component(j))
385 {
386 local_coupling_matrix(i, j) +=
387 (fev.shape_value(i, q)) * inclusion_fe_values[j];
388 }
389 }
390 local_rhs(j) += inclusion_fe_values[j] *
391 inclusions.inclusions_rhs.value(
392 real_q, inclusions.get_component(j));
393
394 local_inclusion_matrix(j, j) +=
395 (inclusion_fe_values[j] * inclusion_fe_values[j] /
396 inclusion_fe_values[0]);
397 }
398 ++p;
399 }
400 // I expect p and next_p to be the same now.
401 Assert(p == next_p, ExcInternalError());
402
403 // Add local matrices to global ones
404 constraints.distribute_local_to_global(local_bulk_matrix,
405 fe_dof_indices,
406 stiffness_matrix);
407
408 constraints.distribute_local_to_global(local_coupling_matrix,
409 fe_dof_indices,
410 inclusion_constraints,
411 inclusion_dof_indices,
412 coupling_matrix);
413 inclusion_constraints.distribute_local_to_global(
414 local_rhs, inclusion_dof_indices, system_rhs.block(1));
415
416 inclusion_constraints.distribute_local_to_global(
417 local_inclusion_matrix, inclusion_dof_indices, inclusion_matrix);
418 }
419 particle = pic.end();
420 }
421 coupling_matrix.compress(VectorOperation::add);
422 stiffness_matrix.compress(VectorOperation::add);
423 inclusion_matrix.compress(VectorOperation::add);
424 system_rhs.compress(VectorOperation::add);
425}

429template <int dim, int spacedim>
430void
431ElasticityProblem<dim, spacedim>::solve()
432{
433 TimerOutput::Scope t(computing_timer, "Solve");
434 LA::MPI::PreconditionAMG prec_A;
435 {
436 LA::MPI::PreconditionAMG::AdditionalData data;
437#ifdef USE_PETSC_LA
438 data.symmetric_operator = true;
439#endif
440 prec_A.initialize(stiffness_matrix, data);
441 }
442
443 const auto A = linear_operator<LA::MPI::Vector>(stiffness_matrix);
444 auto invA = A;
445
446 const auto amgA = linear_operator(A, prec_A);
447
448 SolverCG<LA::MPI::Vector> cg_stiffness(par.inner_control);
449 invA = inverse_operator(A, cg_stiffness, amgA);
450
451 // Some aliases
452 auto &u = solution.block(0);
453 auto &lambda = solution.block(1);
454
455 const auto &f = system_rhs.block(0);
456 const auto &g = system_rhs.block(1);
457
458 if (inclusions.n_dofs() == 0)
459 {
460 u = invA * f;
461 }
462 else
463 {
464 const auto Bt = linear_operator<LA::MPI::Vector>(coupling_matrix);
465 const auto B = transpose_operator(Bt);
466 const auto C = linear_operator<LA::MPI::Vector>(inclusion_matrix);
467
468 // Schur complement
469 const auto S = B * invA * Bt;
470
471 // Schur complement preconditioner
472 auto invS = S;
473 SolverCG<LA::MPI::Vector> cg_schur(par.outer_control);
474 invS = inverse_operator(S, cg_schur);
475
476 pcout << " f norm: " << f.l2_norm() << ", g norm: " << g.l2_norm()
477 << std::endl;
478
479 // Compute Lambda first
480 lambda = invS * (B * invA * f - g);
481 pcout << " Solved for lambda in " << par.outer_control.last_step()
482 << " iterations." << std::endl;
483
484 // Then compute u
485 u = invA * (f - Bt * lambda);
486 pcout << " u norm: " << u.l2_norm()
487 << ", lambda norm: " << lambda.l2_norm() << std::endl;
488 }
489
490 pcout << " Solved for u in " << par.inner_control.last_step()
491 << " iterations." << std::endl;
492 constraints.distribute(u);
493 inclusion_constraints.distribute(lambda);
494 locally_relevant_solution = solution;
495}

499template <int dim, int spacedim>
500void
501ElasticityProblem<dim, spacedim>::refine_and_transfer()
502{
503 TimerOutput::Scope t(computing_timer, "Refine");
504 Vector<float> error_per_cell(tria.n_active_cells());
505 KellyErrorEstimator<spacedim>::estimate(dh,
506 QGauss<spacedim - 1>(par.fe_degree +
507 1),
508 {},
509 locally_relevant_solution.block(0),
510 error_per_cell);
511 if (par.refinement_strategy == "fixed_fraction")
512 {
513 parallel::distributed::GridRefinement::refine_and_coarsen_fixed_fraction(
514 tria, error_per_cell, par.refinement_fraction, par.coarsening_fraction);
515 }
516 else if (par.refinement_strategy == "fixed_number")
517 {
518 parallel::distributed::GridRefinement::refine_and_coarsen_fixed_number(
519 tria,
520 error_per_cell,
521 par.refinement_fraction,
522 par.coarsening_fraction,
523 par.max_cells);
524 }
525 else if (par.refinement_strategy == "global")
526 for (const auto &cell : tria.active_cell_iterators())
527 cell->set_refine_flag();
528
529 parallel::distributed::SolutionTransfer<spacedim, LA::MPI::Vector> transfer(
530 dh);
531 tria.prepare_coarsening_and_refinement();
532 inclusions.inclusions_as_particles.prepare_for_coarsening_and_refinement();
533 transfer.prepare_for_coarsening_and_refinement(
534 locally_relevant_solution.block(0));
535 tria.execute_coarsening_and_refinement();
536 inclusions.inclusions_as_particles.unpack_after_coarsening_and_refinement();
537 setup_dofs();
538 transfer.interpolate(solution.block(0));
539 constraints.distribute(solution.block(0));
540 locally_relevant_solution.block(0) = solution.block(0);
541}

545template <int dim, int spacedim>
546std::string
547ElasticityProblem<dim, spacedim>::output_solution() const
548{
549 TimerOutput::Scope t(computing_timer, "Output results");
550 std::string solution_name = "solution";
551 DataOut<spacedim> data_out;
552 data_out.attach_dof_handler(dh);
553 data_out.add_data_vector(locally_relevant_solution.block(0), solution_name);
554 Vector<float> subdomain(tria.n_active_cells());
555 for (unsigned int i = 0; i < subdomain.size(); ++i)
556 subdomain(i) = tria.locally_owned_subdomain();
557 data_out.add_data_vector(subdomain, "subdomain");
558 data_out.build_patches();
559 const std::string filename =
560 par.output_name + "_" + std::to_string(cycle) + ".vtu";
561 data_out.write_vtu_in_parallel(par.output_directory + "/" + filename,
562 mpi_communicator);
563 return filename;
564}

626template <int dim, int spacedim>
627void
628ElasticityProblem<dim, spacedim>::run()
629{
630 print_parameters();
631 make_grid();
632 setup_fe();
633 inclusions.setup_inclusions_particles(tria);
634 for (cycle = 0; cycle < par.n_refinement_cycles; ++cycle)
635 {
636 setup_dofs();
637 if (par.output_results_before_solving)
638 output_results();
639 assemble_elasticity_system();
640 assemble_coupling();
641 solve();
642 output_results();
643 par.convergence_table.error_from_exact(dh, solution.block(0), par.bc);
644 if (cycle != par.n_refinement_cycles - 1)
645 refine_and_transfer();
646 if (pcout.is_active())
647 par.convergence_table.output_table(pcout.get_stream());
648 }
649}
