57 void
58 initialize()
59 {
60 AssertThrow(n_q_points > 0,
61 ExcMessage(
62 "Refinement of inclusions must be greater than zero."));
63 AssertThrow(n_coefficients > 0,
64 ExcMessage(
65 "Number of coefficients must be greater than zero."));
66 support_points.resize(n_q_points);
67 normals.resize(n_q_points);
68 theta.resize(n_q_points);
69 current_support_points.resize(n_q_points);
70 current_fe_values.resize(n_coefficients);
71
72 for (unsigned int i = 0; i < n_q_points; ++i)
73 {
74 theta[i] = i * 2 * numbers::PI / n_q_points;
75 support_points[i][0] = std::cos(theta[i]);
76 support_points[i][1] = std::sin(theta[i]);
77 normals[i] = support_points[i];
78 }
79
80 if (inclusions_file != "")
81 {
82 std::ifstream infile(inclusions_file);
83 Assert(infile, ExcIO());
84
85 double buffer_double;
86 // cx, cy, R or cx,cy,cz,dx,dy,dz,R
87 const unsigned int size = (spacedim == 2 ? 3 : 7);
88 std::vector<double> inclusion(size);
89
90 while (infile >> buffer_double)
91 {
92 inclusion[0] = buffer_double;
93 for (unsigned int i = 1; i < size; ++i)
94 {
95 Assert(infile, ExcIO());
96 infile >> inclusion[i];
97 }
98 inclusions.push_back(inclusion);
99 }
100 }
101 }

144 const std::vector<double> &
145 get_fe_values(const types::global_dof_index particle_id) const
146 {
147 if (n_coefficients == 0)
148 return current_fe_values;
149 const auto q = particle_id % n_q_points;
150 const auto id = particle_id / n_q_points;
151 AssertIndexRange(id, inclusions.size());
152 const auto r = get_radius(id);
153 const auto ds = 2 * numbers::PI * r / n_q_points * get_direction(id).norm();
154 current_fe_values[0] = ds;
155 for (unsigned int c = 1; c < n_coefficients; ++c)
156 {
157 unsigned int omega = (c + 1) / 2;
158 const double rho = std::pow(r, omega);
159 if ((c + 1) % 2 == 0)
160 current_fe_values[c] = ds * rho * std::cos(theta[q] * omega);
161 else
162 current_fe_values[c] = ds * rho * std::sin(theta[q] * omega);
163 }
164 return current_fe_values;
165 }

202 Tensor<1, spacedim>
203 get_direction(const types::global_dof_index &inclusion_id) const
204 {
205 AssertIndexRange(inclusion_id, inclusions.size());
206 if constexpr (spacedim == 2)
207 {
208 // No direction in 2d. But the norm is used.
209 Tensor<1, spacedim> ret;
210 ret[0] = 1.0;
211 return ret;
212 }
213 else
214 {
215 const auto &inclusion = inclusions[inclusion_id];
216 AssertDimension(inclusion.size(), 2 * spacedim + 1);
217 Tensor<1, spacedim> direction;
218 for (unsigned int d = 0; d < spacedim; ++d)
219 direction[d] = inclusion[spacedim + d];
220 AssertThrow(direction.norm() > 1e-10,
221 ExcMessage("Expecting a direction with non-zero norm"));
222 return direction;

227
228 Tensor<2, spacedim>
229 get_rotation(const types::global_dof_index &inclusion_id) const
230 {
231 Tensor<2, spacedim> rotation = unit_symmetric_tensor<spacedim>();
232 if constexpr (spacedim == 2)
233 {
234 return rotation;
235 }
236 else if constexpr (spacedim == 3)
237 {
238 auto direction = get_direction(inclusion_id);
239 direction /= direction.norm();
240
241 // Build rotation w.r.t. z axis
242 static const auto z_axis = Tensor<1, spacedim>({0, 0, 1});
243 auto v = cross_product_3d(z_axis, direction);
244 const auto cos_t = direction * z_axis;
245
246 if (std::abs(cos_t + 1) < 1e-10)
247 {
248 rotation[1][1] = -1;
249 rotation[2][2] = -1;
250 }
251 else
252 {
253 Tensor<2, spacedim> vx;
254 vx[0] = Tensor<1, spacedim>({0, -v[2], v[1]});
255 vx[1] = Tensor<1, spacedim>({v[2], 0, -v[0]});
256 vx[2] = Tensor<1, spacedim>({-v[1], v[0], 0});
257 auto vx2 = vx * vx;
258 rotation += vx + vx2 * (1 / (1 + cos_t));
259 }
260
261 return rotation;
262 }
