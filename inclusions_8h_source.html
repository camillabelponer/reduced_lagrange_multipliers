154 void
155 initialize()
156 {
157 AssertThrow(n_q_points > 0,
158 ExcMessage(
159 "Refinement of inclusions must be greater than zero."));
160 AssertThrow(n_coefficients > 0,
161 ExcMessage(
162 "Number of coefficients must be greater than zero."));
163 support_points.resize(n_q_points);
164 normals.resize(n_q_points);
165 theta.resize(n_q_points);
166 current_support_points.resize(n_q_points);
167
168 for (unsigned int i = 0; i < n_q_points; ++i)
169 {
170 theta[i] = i * 2 * numbers::PI / n_q_points;
171 support_points[i][0] = std::cos(theta[i]);
172 support_points[i][1] = std::sin(theta[i]);
173 normals[i] = support_points[i];
174 }
175
176 // Make sure that selected coefficients is the iota vector, when we don't
177 // select anything for backward compatibility.
178 if (selected_coefficients.empty())
179 {
180 selected_coefficients.resize(n_coefficients * n_vector_components);
181 for (unsigned int i = 0; i < n_coefficients * n_vector_components; ++i)
182 selected_coefficients[i] = i;
183 }
184 else
185 {
186 // This is zero when we use the selection, since it is not used.
187 coefficient_offset = 0;
188 }
189
190 // This MUST be here, otherwise n_dofs_per_inclusions() would be wrong.
191 current_fe_values.resize(n_dofs_per_inclusion());
192
193 if (inclusions_file != "")
194 {
195 std::ifstream infile(inclusions_file);
196 Assert(infile, ExcIO());
197
198 double buffer_double;
199 // cx, cy, R or cx,cy,cz,dx,dy,dz,R,vesselID
200 const unsigned int size = (spacedim == 2 ? 3 : 8);
201 std::vector<double> inclusion(size);
202
203 while (infile >> buffer_double)
204 {
205 inclusion[0] = buffer_double;
206 for (unsigned int i = 1; i < size; ++i)
207 {
208 Assert(infile, ExcIO());
209 infile >> inclusion[i];
210 }
211 inclusions.push_back(inclusion);
212 }
213 }
214
215 if (data_file != "")
216 {
217 std::ifstream infile(data_file);
218 Assert(infile, ExcIO());
219
220 std::string line;
221 while (std::getline(infile, line))
222 {
223 std::vector<double> double_line;
224 std::istringstream iss(line);
225 double buffer_double;
226 while (iss >> buffer_double)
227 {
228 double_line.push_back(buffer_double);
229 }
230 inclusions_data.push_back(double_line);
231 }
232 AssertThrow(inclusions_data.size() == n_inclusions(),
233 ExcDimensionMismatch(inclusions_data.size(),
234 n_inclusions()));
235 if (inclusions_data.size() > 0)
236 {
237 const auto N = inclusions_data[0].size();
238 for (const auto &l : inclusions_data)
239 {
240 AssertThrow(l.size() == N, ExcDimensionMismatch(l.size(), N));
241 }
242 if (Utilities::MPI::this_mpi_process(mpi_communicator) == 0)
243 {
244 std::cout << "rank "
245 << Utilities::MPI::this_mpi_process(mpi_communicator)
246 << ": Read " << N << " coefficients per "
247 << inclusions.size() << " inclusion" << std::endl;
248 }
249 }
250 }
251 check_vessels();
252 }

585 Tensor<2, spacedim>
586 get_rotation(const types::global_dof_index &inclusion_id) const
587 {
588 Tensor<2, spacedim> rotation = unit_symmetric_tensor<spacedim>();
589 if constexpr (spacedim == 2)
590 {
591 return rotation;
592 }
593 else if constexpr (spacedim == 3)
594 {
595 auto direction = get_direction(inclusion_id);
596 direction /= direction.norm();
597
598 // Build rotation w.r.t. z axis
599 static const auto z_axis = Tensor<1, spacedim>({0, 0, 1});
600 auto v = cross_product_3d(z_axis, direction);
601 const auto cos_t = direction * z_axis;
602
603 if (std::abs(cos_t + 1) < 1e-10)
604 {
605 rotation[1][1] = -1;
606 rotation[2][2] = -1;
607 }
608 else
609 {
610 Tensor<2, spacedim> vx;
611 vx[0] = Tensor<1, spacedim>({0, -v[2], v[1]});
612 vx[1] = Tensor<1, spacedim>({v[2], 0, -v[0]});
613 vx[2] = Tensor<1, spacedim>({-v[1], v[0], 0});
614 auto vx2 = vx * vx;
615 rotation += vx + vx2 * (1 / (1 + cos_t));
616 }
617
618 return rotation;
619 }
