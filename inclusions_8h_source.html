76 void
77 initialize()
78 {
79 AssertThrow(n_q_points > 0,
80 ExcMessage(
81 "Refinement of inclusions must be greater than zero."));
82 AssertThrow(n_coefficients > 0,
83 ExcMessage(
84 "Number of coefficients must be greater than zero."));
85 support_points.resize(n_q_points);
86 normals.resize(n_q_points);
87 theta.resize(n_q_points);
88 current_support_points.resize(n_q_points);
89 current_fe_values.resize(n_dofs_per_inclusion());
90
91 for (unsigned int i = 0; i < n_q_points; ++i)
92 {
93 theta[i] = i * 2 * numbers::PI / n_q_points;
94 support_points[i][0] = std::cos(theta[i]);
95 support_points[i][1] = std::sin(theta[i]);
96 normals[i] = support_points[i];
97 }
98
99 if (inclusions_file != "")
100 {
101 std::ifstream infile(inclusions_file);
102 Assert(infile, ExcIO());
103
104 double buffer_double;
105 // cx, cy, R or cx,cy,cz,dx,dy,dz,R
106 const unsigned int size = (spacedim == 2 ? 3 : 7);
107 std::vector<double> inclusion(size);
108
109 while (infile >> buffer_double)
110 {
111 inclusion[0] = buffer_double;
112 for (unsigned int i = 1; i < size; ++i)
113 {
114 Assert(infile, ExcIO());
115 infile >> inclusion[i];
116 }
117 inclusions.push_back(inclusion);
118 }
119 }
120 }

135 setup_inclusions_particles(
136 const parallel::distributed::Triangulation<spacedim> &tria)
137 {
138 initialize();
139 mpi_communicator = tria.get_communicator();
140 inclusions_as_particles.initialize(tria,
141 StaticMappingQ1<spacedim>::mapping);
142
143 if (n_dofs() == 0)
144 return;
145
146 std::vector<Point<spacedim>> particles_positions;
147 particles_positions.reserve(n_particles());
148 for (unsigned int i = 0; i < n_inclusions(); ++i)
149 {
150 const auto &p = get_current_support_points(i);
151 particles_positions.insert(particles_positions.end(),
152 p.begin(),
153 p.end());
154 }
155
156 std::vector<BoundingBox<spacedim>> all_boxes;
157 all_boxes.reserve(tria.n_locally_owned_active_cells());
158 for (const auto &cell : tria.active_cell_iterators())
159 if (cell->is_locally_owned())
160 all_boxes.emplace_back(cell->bounding_box());
161 const auto tree = pack_rtree(all_boxes);
162 const auto local_boxes = extract_rtree_level(tree, rtree_extraction_level);
163
164 auto global_bounding_boxes =
165 Utilities::MPI::all_gather(mpi_communicator, local_boxes);
166
167 Assert(!global_bounding_boxes.empty(),
168 ExcInternalError(
169 "I was expecting the "
170 "global_bounding_boxes to be filled at this stage. "
171 "Make sure you fill this vector before trying to use it "
172 "here. Bailing out."));
173 inclusions_as_particles.insert_global_particles(particles_positions,
174 global_bounding_boxes);
175 // pcout << "Inclusions particles: "
176 // << inclusions_as_particles.n_global_particles() << std::endl;
177 }

230 const std::vector<double> &
231 get_fe_values(const types::global_dof_index particle_id) const
232 {
233 if (n_coefficients == 0)
234 return current_fe_values;
235 const auto q = particle_id % n_q_points;
236 const auto id = particle_id / n_q_points;
237 AssertIndexRange(id, inclusions.size());
238 const auto r = get_radius(id);
239 const auto ds = 2 * numbers::PI * r / n_q_points * get_direction(id).norm();
240 current_fe_values[0] = ds;
241 for (unsigned int c = 1; c < n_coefficients; ++c)
242 {
243 unsigned int omega = (c + 1) / 2;
244 const double rho = std::pow(r, omega);
245 for (unsigned int i = 0; i < n_vector_components; ++i)
246 if ((c + 1) % 2 == 0)
247 current_fe_values[c * n_vector_components + i] =
248 ds * rho * std::cos(theta[q] * omega);
249 else
250 current_fe_values[c * n_vector_components + i] =
251 ds * rho * std::sin(theta[q] * omega);
252 }
253 return current_fe_values;
254 }

296 Tensor<1, spacedim>
297 get_direction(const types::global_dof_index &inclusion_id) const
298 {
299 AssertIndexRange(inclusion_id, inclusions.size());
300 if constexpr (spacedim == 2)
301 {
302 // No direction in 2d. But the norm is used.
303 Tensor<1, spacedim> ret;
304 ret[0] = 1.0;
305 return ret;
306 }
307 else
308 {
309 const auto &inclusion = inclusions[inclusion_id];
310 AssertDimension(inclusion.size(), 2 * spacedim + 1);
311 Tensor<1, spacedim> direction;
312 for (unsigned int d = 0; d < spacedim; ++d)
313 direction[d] = inclusion[spacedim + d];
314 AssertThrow(direction.norm() > 1e-10,
315 ExcMessage("Expecting a direction with non-zero norm"));
316 return direction;
317 }

321
322 Tensor<2, spacedim>
323 get_rotation(const types::global_dof_index &inclusion_id) const
324 {
325 Tensor<2, spacedim> rotation = unit_symmetric_tensor<spacedim>();
326 if constexpr (spacedim == 2)
327 {
328 return rotation;
329 }
330 else if constexpr (spacedim == 3)
331 {
332 auto direction = get_direction(inclusion_id);
333 direction /= direction.norm();
334
335 // Build rotation w.r.t. z axis
336 static const auto z_axis = Tensor<1, spacedim>({0, 0, 1});
337 auto v = cross_product_3d(z_axis, direction);
338 const auto cos_t = direction * z_axis;
339
340 if (std::abs(cos_t + 1) < 1e-10)
341 {
342 rotation[1][1] = -1;
343 rotation[2][2] = -1;
344 }
345 else
346 {
347 Tensor<2, spacedim> vx;
348 vx[0] = Tensor<1, spacedim>({0, -v[2], v[1]});
349 vx[1] = Tensor<1, spacedim>({v[2], 0, -v[0]});
350 vx[2] = Tensor<1, spacedim>({-v[1], v[0], 0});
351 auto vx2 = vx * vx;
352 rotation += vx + vx2 * (1 / (1 + cos_t));
353 }
354
355 return rotation;
356 }
