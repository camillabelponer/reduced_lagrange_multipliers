158 void
159 initialize()
160 {
161 AssertThrow(n_q_points > 0,
162 ExcMessage(
163 "Refinement of inclusions must be greater than zero."));
164 AssertThrow(n_coefficients > 0,
165 ExcMessage(
166 "Number of coefficients must be greater than zero."));
167 support_points.resize(n_q_points);
168 normals.resize(n_q_points);
169 theta.resize(n_q_points);
170 current_support_points.resize(n_q_points);
171
172 for (unsigned int i = 0; i < n_q_points; ++i)
173 {
174 theta[i] = i * 2 * numbers::PI / n_q_points;
175 support_points[i][0] = std::cos(theta[i]);
176 support_points[i][1] = std::sin(theta[i]);
177 normals[i] = support_points[i];
178 }
179
180 // Make sure that selected coefficients is the iota vector, when we don't
181 // select anything for backward compatibility.
182 if (selected_coefficients.empty())
183 {
184 selected_coefficients.resize(n_coefficients * n_vector_components);
185 for (unsigned int i = 0; i < n_coefficients * n_vector_components; ++i)
186 selected_coefficients[i] = i;
187 }
188 else
189 {
190 // This is zero when we use the selection, since it is not used.
191 coefficient_offset = 0;
192 }
193
194 // This MUST be here, otherwise n_dofs_per_inclusions() would be wrong.
195 current_fe_values.resize(n_dofs_per_inclusion());
196
197 if (inclusions_file != "")
198 {
199 std::ifstream infile(inclusions_file);
200 Assert(infile, ExcIO());
201
202 double buffer_double;
203 // cx, cy, R or cx,cy,cz,dx,dy,dz,R,vesselID
204 const unsigned int size = (spacedim == 2 ? 3 : 8);
205 std::vector<double> inclusion(size);
206
207 while (infile >> buffer_double)
208 {
209 inclusion[0] = buffer_double;
210 for (unsigned int i = 1; i < size; ++i)
211 {
212 Assert(infile, ExcIO());
213 infile >> inclusion[i];
214 }
215 inclusions.push_back(inclusion);
216 }
217 }
218
219 if (data_file != "")
220 {
221 std::ifstream infile(data_file);
222 Assert(infile, ExcIO());
223
224 std::string line;
225 while (std::getline(infile, line))
226 {
227 std::vector<double> double_line;
228 std::istringstream iss(line);
229 double buffer_double;
230 while (iss >> buffer_double)
231 {
232 double_line.push_back(buffer_double);
233 }
234 inclusions_data.push_back(double_line);
235 }
236 AssertThrow(inclusions_data.size() == n_inclusions(),
237 ExcDimensionMismatch(inclusions_data.size(),
238 n_inclusions()));
239 if (inclusions_data.size() > 0)
240 {
241 const auto N = inclusions_data[0].size();
242 for (const auto &l : inclusions_data)
243 {
244 AssertThrow(l.size() == N, ExcDimensionMismatch(l.size(), N));
245 }
246 if (Utilities::MPI::this_mpi_process(mpi_communicator) == 0)
247 {
248 std::cout << "rank "
249 << Utilities::MPI::this_mpi_process(mpi_communicator)
250 << ": Read " << N << " coefficients per "
251 << inclusions.size() << " inclusion" << std::endl;
252 }
253 }
254 }
255 check_vessels();
256 }

589 Tensor<2, spacedim>
590 get_rotation(const types::global_dof_index &inclusion_id) const
591 {
592 Tensor<2, spacedim> rotation = unit_symmetric_tensor<spacedim>();
593 if constexpr (spacedim == 2)
594 {
595 return rotation;
596 }
597 else if constexpr (spacedim == 3)
598 {
599 auto direction = get_direction(inclusion_id);
600 direction /= direction.norm();
601
602 // Build rotation w.r.t. z axis
603 static const auto z_axis = Tensor<1, spacedim>({0, 0, 1});
604 auto v = cross_product_3d(z_axis, direction);
605 const auto cos_t = direction * z_axis;
606
607 if (std::abs(cos_t + 1) < 1e-10)
608 {
609 rotation[1][1] = -1;
610 rotation[2][2] = -1;
611 }
612 else
613 {
614 Tensor<2, spacedim> vx;
615 vx[0] = Tensor<1, spacedim>({0, -v[2], v[1]});
616 vx[1] = Tensor<1, spacedim>({v[2], 0, -v[0]});
617 vx[2] = Tensor<1, spacedim>({-v[1], v[0], 0});
618 auto vx2 = vx * vx;
619 rotation += vx + vx2 * (1 / (1 + cos_t));
620 }
621
622 return rotation;
623 }
