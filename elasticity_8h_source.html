121template <int dim, int spacedim>
122ElasticityProblemParameters<dim, spacedim>::ElasticityProblemParameters()
123 : ParameterAcceptor("/Immersed Problem/")
124 , rhs("/Immersed Problem/Right hand side", spacedim)
125 , bc("/Immersed Problem/Dirichlet boundary conditions", spacedim)
126 , inner_control("/Immersed Problem/Solver/Inner control")
127 , outer_control("/Immersed Problem/Solver/Outer control")
128 , convergence_table(std::vector<std::string>(spacedim, "u"),
129 std::vector<std::set<VectorTools::NormType>>(
130 spacedim,
131 {VectorTools::L2_norm, VectorTools::H1_norm}))
132{
133 add_parameter("FE degree", fe_degree, "", this->prm, Patterns::Integer(1));
134 add_parameter("Output directory", output_directory);
135 add_parameter("Output name", output_name);
136 add_parameter("Output results also before solving",
137 output_results_before_solving);
138 add_parameter("Initial refinement", initial_refinement);
139 add_parameter("Dirichlet boundary ids", dirichlet_ids);
140 enter_subsection("Grid generation");
141 {
142 add_parameter("Grid generator", name_of_grid);
143 add_parameter("Grid generator arguments", arguments_for_grid);
144 }
145 leave_subsection();
146 enter_subsection("Refinement and remeshing");
147 {
148 add_parameter("Strategy",
149 refinement_strategy,
150 "",
151 this->prm,
152 Patterns::Selection("fixed_fraction|fixed_number|global"));
153 add_parameter("Coarsening fraction", coarsening_fraction);
154 add_parameter("Refinement fraction", refinement_fraction);
155 add_parameter("Maximum number of cells", max_cells);
156 add_parameter("Number of refinement cycles", n_refinement_cycles);
157 }
158 leave_subsection();
159
160 this->prm.enter_subsection("Error");
161 convergence_table.add_parameters(this->prm);
162 this->prm.leave_subsection();
163}
