61template <int dim, int spacedim>
62void
63read_grid_and_cad_files(const std::string &grid_file_name,
64 const std::string &ids_and_cad_file_names,
65 Triangulation<dim, spacedim> &tria)
66{
67 GridIn<dim, spacedim> grid_in;
68 grid_in.attach_triangulation(tria);
69 grid_in.read(grid_file_name);
70#ifdef DEAL_II_WITH_OPENCASCADE
71 using map_type = std::map<types::manifold_id, std::string>;
72 using Converter = Patterns::Tools::Convert<map_type>;
73 for (const auto &pair : Converter::to_value(ids_and_cad_file_names))
74 {
75 const auto &manifold_id = pair.first;
76 const auto &cad_file_name = pair.second;
77 const auto extension = to_lower_copy(
78 cad_file_name.substr(cad_file_name.find_last_of('.') + 1));
79 TopoDS_Shape shape;
80 if (extension == "iges" || extension == "igs")
81 shape = OpenCASCADE::read_IGES(cad_file_name);
82 else if (extension == "step" || extension == "stp")
83 shape = OpenCASCADE::read_STEP(cad_file_name);
84 else
85 AssertThrow(false,
86 ExcNotImplemented("We found an extension that we "
87 "do not recognize as a CAD file "
88 "extension. Bailing out."));
89 const auto n_elements = OpenCASCADE::count_elements(shape);
90 if ((std::get<0>(n_elements) == 0))
91 tria.set_manifold(
92 manifold_id,
93 OpenCASCADE::ArclengthProjectionLineManifold<dim, spacedim>(shape));
94 else if (spacedim == 3)
95 {
96 const auto t = reinterpret_cast<Triangulation<dim, 3> *>(&tria);
97 t->set_manifold(manifold_id,
98 OpenCASCADE::NormalToMeshProjectionManifold<dim, 3>(
99 shape));
100 }
101 else
102 tria.set_manifold(manifold_id,
103 OpenCASCADE::NURBSPatchManifold<dim, spacedim>(
104 TopoDS::Face(shape)));
105 }
106#else
107 (void)ids_and_cad_file_names;
108 AssertThrow(false, ExcNotImplemented("Generation of the grid failed."));
109#endif
110}

145template <int dim, int spacedim>
146void
147PoissonProblem<dim, spacedim>::setup_dofs()
148{
149 TimerOutput::Scope t(computing_timer, "Setup dofs");
150 dh.distribute_dofs(*fe);
151#ifdef MATRIX_FREE_PATH
152 dh.distribute_mg_dofs();
153#endif
154
155 owned_dofs.resize(2);
156 owned_dofs[0] = dh.locally_owned_dofs();
157 relevant_dofs.resize(2);
158 DoFTools::extract_locally_relevant_dofs(dh, relevant_dofs[0]);
159 {
160 constraints.reinit(owned_dofs[0], relevant_dofs[0]);
161 DoFTools::make_hanging_node_constraints(dh, constraints);
162 for (const auto id : par.dirichlet_ids)
163 VectorTools::interpolate_boundary_values(dh, id, par.bc, constraints);
164 constraints.close();
165 }
166 {
167#ifdef MATRIX_FREE_PATH
168
169 typename MatrixFree<spacedim, double>::AdditionalData additional_data;
170 additional_data.tasks_parallel_scheme =
171 MatrixFree<spacedim, double>::AdditionalData::none;
172 additional_data.mapping_update_flags =
173 (update_gradients | update_JxW_values | update_quadrature_points);
174 std::shared_ptr<MatrixFree<spacedim, double>> system_mf_storage(
175 new MatrixFree<spacedim, double>());
176 system_mf_storage->reinit(
177 mapping, dh, constraints, QGauss<1>(fe->degree + 1), additional_data);
178 stiffness_matrix.initialize(system_mf_storage);
179
180 // Perform setup for matrix-free multigrid
181 {
182 const unsigned int nlevels = tria.n_global_levels();
183 mg_matrices.resize(0, nlevels - 1);
184
185 const std::set<types::boundary_id> dirichlet_boundary_ids = {0};
186 mg_constrained_dofs.initialize(dh);
187 mg_constrained_dofs.make_zero_boundary_constraints(
188 dh, dirichlet_boundary_ids);
189
190 for (unsigned int level = 0; level < nlevels; ++level)
191 {
192 AffineConstraints<double> level_constraints(
193 dh.locally_owned_mg_dofs(level),
194 DoFTools::extract_locally_relevant_level_dofs(dh, level));
195 for (const types::global_dof_index dof_index :
196 mg_constrained_dofs.get_boundary_indices(level))
197 level_constraints.constrain_dof_to_zero(dof_index);
198 level_constraints.close();
199
200 typename MatrixFree<spacedim, float>::AdditionalData
201 additional_data_level;
202 additional_data_level.tasks_parallel_scheme =
203 MatrixFree<spacedim, float>::AdditionalData::none;
204 additional_data_level.mapping_update_flags =
205 (update_gradients | update_JxW_values | update_quadrature_points);
206 additional_data_level.mg_level = level;
207 std::shared_ptr<MatrixFree<spacedim, float>> mg_mf_storage_level =
208 std::make_shared<MatrixFree<spacedim, float>>();
209 mg_mf_storage_level->reinit(mapping,
210 dh,
211 level_constraints,
212 QGauss<1>(fe->degree + 1),
213 additional_data_level);
214
215 mg_matrices[level].initialize(mg_mf_storage_level,
216 mg_constrained_dofs,
217 level);
218 }
219 }
220
221
222#else
223
224 stiffness_matrix.clear();
225 DynamicSparsityPattern dsp(relevant_dofs[0]);
226 DoFTools::make_sparsity_pattern(dh, dsp, constraints, false);
227 SparsityTools::distribute_sparsity_pattern(dsp,
228 owned_dofs[0],
229 mpi_communicator,
230 relevant_dofs[0]);
231 stiffness_matrix.reinit(owned_dofs[0],
232 owned_dofs[0],
233 dsp,
234 mpi_communicator);
235
236#endif
237 }
238 inclusion_constraints.close();
239
240 if (inclusions.n_dofs() > 0)
241 {
242 auto inclusions_set =
243 Utilities::MPI::create_evenly_distributed_partitioning(
244 mpi_communicator, inclusions.n_inclusions());
245
246 owned_dofs[1] = inclusions_set.tensor_product(
247 complete_index_set(inclusions.n_coefficients));
248
249 coupling_matrix.clear();
250 DynamicSparsityPattern dsp(dh.n_dofs(),
251 inclusions.n_dofs(),
252 relevant_dofs[0]);
253
254 relevant_dofs[1] = assemble_coupling_sparsity(dsp);
255 SparsityTools::distribute_sparsity_pattern(dsp,
256 owned_dofs[0],
257 mpi_communicator,
258 relevant_dofs[0]);
259 coupling_matrix.reinit(owned_dofs[0],
260 owned_dofs[1],
261 dsp,
262 mpi_communicator);
263
264 DynamicSparsityPattern idsp(inclusions.n_dofs(),
265 inclusions.n_dofs(),
266 relevant_dofs[1]);
267 for (const auto i : relevant_dofs[1])
268 idsp.add(i, i);
269
270 SparsityTools::distribute_sparsity_pattern(idsp,
271 owned_dofs[1],
272 mpi_communicator,
273 relevant_dofs[1]);
274 inclusion_matrix.reinit(owned_dofs[1],
275 owned_dofs[1],
276 idsp,
277 mpi_communicator);
278 }
279
280 locally_relevant_solution.reinit(owned_dofs, relevant_dofs, mpi_communicator);
281
282#ifdef MATRIX_FREE_PATH
283 system_rhs.reinit(owned_dofs, relevant_dofs, mpi_communicator);
284 solution.reinit(owned_dofs, relevant_dofs, mpi_communicator);
285#else
286 system_rhs.reinit(owned_dofs, mpi_communicator);
287 solution.reinit(owned_dofs, mpi_communicator);
288#endif
289
290 pcout << " Number of degrees of freedom: " << owned_dofs[0].size() << " + "
291 << owned_dofs[1].size()
292 << " (locally owned: " << owned_dofs[0].n_elements() << " + "
293 << owned_dofs[1].n_elements() << ")" << std::endl;
294}

298template <int dim, int spacedim>
299void
300PoissonProblem<dim, spacedim>::assemble_poisson_system()
301{
302 stiffness_matrix = 0;
303 coupling_matrix = 0;
304 system_rhs = 0;
305 FEValues<spacedim> fe_values(*fe,
306 *quadrature,
307 update_values | update_gradients |
308 update_quadrature_points | update_JxW_values);
309 const unsigned int dofs_per_cell = fe->n_dofs_per_cell();
310 const unsigned int n_q_points = quadrature->size();
311 FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
312 Vector<double> cell_rhs(dofs_per_cell);
313 std::vector<double> rhs_values(n_q_points);
314 std::vector<Tensor<1, spacedim>> grad_phi_u(dofs_per_cell);
315 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
316 const FEValuesExtractors::Scalar scalar(0);
317 for (const auto &cell : dh.active_cell_iterators())
318 if (cell->is_locally_owned())
319 {
320 cell_matrix = 0;
321 cell_rhs = 0;
322 fe_values.reinit(cell);
323 par.rhs.value_list(fe_values.get_quadrature_points(), rhs_values);
324 for (unsigned int q = 0; q < n_q_points; ++q)
325 {
326 for (unsigned int k = 0; k < dofs_per_cell; ++k)
327 grad_phi_u[k] = fe_values[scalar].gradient(k, q);
328 for (unsigned int i = 0; i < dofs_per_cell; ++i)
329 {
330 for (unsigned int j = 0; j < dofs_per_cell; ++j)
331 {
332 cell_matrix(i, j) +=
333 grad_phi_u[i] * grad_phi_u[j] * fe_values.JxW(q);
334 }
335 cell_rhs(i) += fe_values.shape_value(i, q) * rhs_values[q] *
336 fe_values.JxW(q);
337 }
338 }
339 cell->get_dof_indices(local_dof_indices);
340 constraints.distribute_local_to_global(cell_matrix,
341 cell_rhs,
342 local_dof_indices,
343 stiffness_matrix,
344 system_rhs.block(0));
345 }
346 stiffness_matrix.compress(VectorOperation::add);
347 system_rhs.compress(VectorOperation::add);
348}
349#endif

352template <int dim, int spacedim>
353IndexSet
354PoissonProblem<dim, spacedim>::assemble_coupling_sparsity(
355 DynamicSparsityPattern &dsp) const
356{
357 TimerOutput::Scope t(computing_timer, "Assemble Coupling sparsity");
358 IndexSet relevant(inclusions.n_dofs());
359
360 const FEValuesExtractors::Scalar scalar(0);
361
362 std::vector<types::global_dof_index> dof_indices(fe->n_dofs_per_cell());
363 std::vector<types::global_dof_index> inclusion_dof_indices;
364
365 auto particle = inclusions.inclusions_as_particles.begin();
366 while (particle != inclusions.inclusions_as_particles.end())
367 {
368 const auto &cell = particle->get_surrounding_cell();
369 const auto dh_cell =
370 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
371 dh_cell->get_dof_indices(dof_indices);
372
373 const auto pic =
374 inclusions.inclusions_as_particles.particles_in_cell(cell);
375 Assert(pic.begin() == particle, ExcInternalError());
376 std::set<types::global_dof_index> inclusion_dof_indices_set;
377 for (const auto &p : pic)
378 {
379 const auto ids = inclusions.get_dof_indices(p.get_id());
380 inclusion_dof_indices_set.insert(ids.begin(), ids.end());
381 }
382 inclusion_dof_indices.resize(0);
383 inclusion_dof_indices.insert(inclusion_dof_indices.begin(),
384 inclusion_dof_indices_set.begin(),
385 inclusion_dof_indices_set.end());
386
387 constraints.add_entries_local_to_global(dof_indices,
388 inclusion_constraints,
389 inclusion_dof_indices,
390 dsp);
391
392 relevant.add_indices(inclusion_dof_indices.begin(),
393 inclusion_dof_indices.end());
394 particle = pic.end();
395 }
396 return relevant;
397}

401template <int dim, int spacedim>
402void
403PoissonProblem<dim, spacedim>::assemble_coupling()
404{
405 TimerOutput::Scope t(computing_timer, "Assemble Coupling matrix");
406
407 const FEValuesExtractors::Scalar scalar(0);
408 std::vector<types::global_dof_index> fe_dof_indices(fe->n_dofs_per_cell());
409 std::vector<types::global_dof_index> inclusion_dof_indices(
410 inclusions.n_coefficients);
411
412 FullMatrix<double> local_coupling_matrix(fe->n_dofs_per_cell(),
413 inclusions.n_coefficients);
414
415 [[maybe_unused]] FullMatrix<double> local_bulk_matrix(fe->n_dofs_per_cell(),
416 fe->n_dofs_per_cell());
417
418 FullMatrix<double> local_inclusion_matrix(inclusions.n_coefficients,
419 inclusions.n_coefficients);
420
421 Vector<double> local_rhs(inclusions.n_coefficients);
422
423 auto particle = inclusions.inclusions_as_particles.begin();
424 while (particle != inclusions.inclusions_as_particles.end())
425 {
426 const auto &cell = particle->get_surrounding_cell();
427 const auto dh_cell =
428 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
429 dh_cell->get_dof_indices(fe_dof_indices);
430 const auto pic =
431 inclusions.inclusions_as_particles.particles_in_cell(cell);
432 Assert(pic.begin() == particle, ExcInternalError());
433
434 auto p = pic.begin();
435 auto next_p = pic.begin();
436 while (p != pic.end())
437 {
438 const auto inclusion_id = inclusions.get_inclusion_id(p->get_id());
439 inclusion_dof_indices = inclusions.get_dof_indices(p->get_id());
440 local_coupling_matrix = 0;
441 local_inclusion_matrix = 0;
442 local_rhs = 0;
443
444 // Extract all points that refer to the same inclusion
445 std::vector<Point<spacedim>> ref_q_points;
446 for (; next_p != pic.end() &&
447 inclusions.get_inclusion_id(next_p->get_id()) == inclusion_id;
448 ++next_p)
449 ref_q_points.push_back(next_p->get_reference_location());
450 FEValues<spacedim, spacedim> fev(*fe,
451 ref_q_points,
452 update_values | update_gradients);
453 fev.reinit(dh_cell);
454 for (unsigned int q = 0; q < ref_q_points.size(); ++q)
455 {
456 const auto id = p->get_id();
457 const auto &inclusion_fe_values = inclusions.get_fe_values(id);
458 const auto &real_q = p->get_location();
459
460 // Coupling and inclusions matrix
461 for (unsigned int j = 0; j < inclusions.n_coefficients; ++j)
462 {
463 for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
464 local_coupling_matrix(i, j) +=
465 (fev.shape_value(i, q)) * inclusion_fe_values[j];
466 local_rhs(j) += inclusion_fe_values[j] *
467 inclusions.inclusions_rhs.value(real_q);
468
469 local_inclusion_matrix(j, j) +=
470 (inclusion_fe_values[j] * inclusion_fe_values[j] /
471 inclusion_fe_values[0]);
472 }
473 ++p;
474 }
475 // I expect p and next_p to be the same now.
476 Assert(p == next_p, ExcInternalError());
477
478 // Add local matrices to global ones
479 constraints.distribute_local_to_global(local_coupling_matrix,
480 fe_dof_indices,
481 inclusion_constraints,
482 inclusion_dof_indices,
483 coupling_matrix);
484 inclusion_constraints.distribute_local_to_global(
485 local_rhs, inclusion_dof_indices, system_rhs.block(1));
486
487 inclusion_constraints.distribute_local_to_global(
488 local_inclusion_matrix, inclusion_dof_indices, inclusion_matrix);
489 }
490 particle = pic.end();
491 }
492 coupling_matrix.compress(VectorOperation::add);
493#ifndef MATRIX_FREE_PATH
494 stiffness_matrix.compress(VectorOperation::add);
495#endif
496 inclusion_matrix.compress(VectorOperation::add);
497 system_rhs.compress(VectorOperation::add);
498}

501#ifdef MATRIX_FREE_PATH
502template <int dim, int spacedim>
503void
504PoissonProblem<dim, spacedim>::assemble_rhs()
505{
506 stiffness_matrix.get_matrix_free()->initialize_dof_vector(solution.block(0));
507 stiffness_matrix.get_matrix_free()->initialize_dof_vector(
508 system_rhs.block(0));
509 system_rhs = 0;
510 solution.block(0) = 0;
511 constraints.distribute(solution.block(0));
512 solution.block(0).update_ghost_values();
513
514 FEEvaluation<spacedim, -1> phi(*stiffness_matrix.get_matrix_free());
515 for (unsigned int cell = 0;
516 cell < stiffness_matrix.get_matrix_free()->n_cell_batches();
517 ++cell)
518 {
519 phi.reinit(cell);
520 phi.read_dof_values_plain(solution.block(0));
521 phi.evaluate(EvaluationFlags::gradients);
522 for (unsigned int q = 0; q < phi.n_q_points; ++q)
523 {
524 const Point<spacedim, VectorizedArray<double>> p_vect =
525 phi.quadrature_point(q);
526
527 VectorizedArray<double> f_value = 0.0;
528 for (unsigned int v = 0; v < VectorizedArray<double>::size(); ++v)
529 {
530 Point<spacedim> p;
531 for (unsigned int d = 0; d < spacedim; ++d)
532 p[d] = p_vect[d][v];
533 f_value[v] = par.rhs.value(p);
534 }
535 phi.submit_gradient(-phi.get_gradient(q), q);
536 phi.submit_value(f_value, q);
537 }
538 phi.integrate(EvaluationFlags::values | EvaluationFlags::gradients);
539 phi.distribute_local_to_global(system_rhs.block(0));
540 }
541
542 system_rhs.compress(VectorOperation::add);
543}
544#endif

547template <int dim, int spacedim>
548void
549PoissonProblem<dim, spacedim>::solve()
550{
551 TimerOutput::Scope t(computing_timer, "Solve");
552
553 SolverCG<VectorType> cg_stiffness(par.inner_control);
554#ifdef MATRIX_FREE_PATH
555
556 using Payload = dealii::internal::LinearOperatorImplementation::EmptyPayload;
557 LinearOperator<VectorType, VectorType, Payload> A;
558 A = linear_operator<VectorType, VectorType, Payload>(stiffness_matrix);
559
560 MGTransferMatrixFree<spacedim, float> mg_transfer(mg_constrained_dofs);
561 mg_transfer.build(dh);
562
563 using SmootherType =
564 PreconditionChebyshev<LevelMatrixType,
565 LinearAlgebra::distributed::Vector<float>>;
566 mg::SmootherRelaxation<SmootherType,
567 LinearAlgebra::distributed::Vector<float>>
568 mg_smoother;
569 MGLevelObject<typename SmootherType::AdditionalData> smoother_data;
570 smoother_data.resize(0, tria.n_global_levels() - 1);
571 for (unsigned int level = 0; level < tria.n_global_levels(); ++level)
572 {
573 if (level > 0)
574 {
575 smoother_data[level].smoothing_range = 15.;
576 smoother_data[level].degree = 5;
577 smoother_data[level].eig_cg_n_iterations = 10;
578 }
579 else
580 {
581 smoother_data[0].smoothing_range = 1e-3;
582 smoother_data[0].degree = numbers::invalid_unsigned_int;
583 smoother_data[0].eig_cg_n_iterations = mg_matrices[0].m();
584 }
585 mg_matrices[level].compute_diagonal();
586 smoother_data[level].preconditioner =
587 mg_matrices[level].get_matrix_diagonal_inverse();
588 }
589 mg_smoother.initialize(mg_matrices, smoother_data);
590
591 MGCoarseGridApplySmoother<LinearAlgebra::distributed::Vector<float>>
592 mg_coarse;
593 mg_coarse.initialize(mg_smoother);
594
595 mg::Matrix<LinearAlgebra::distributed::Vector<float>> mg_matrix(mg_matrices);
596
597 MGLevelObject<MatrixFreeOperators::MGInterfaceOperator<LevelMatrixType>>
598 mg_interface_matrices;
599 mg_interface_matrices.resize(0, tria.n_global_levels() - 1);
600 for (unsigned int level = 0; level < tria.n_global_levels(); ++level)
601 mg_interface_matrices[level].initialize(mg_matrices[level]);
602 mg::Matrix<LinearAlgebra::distributed::Vector<float>> mg_interface(
603 mg_interface_matrices);
604
605 Multigrid<LinearAlgebra::distributed::Vector<float>> mg(
606 mg_matrix, mg_coarse, mg_transfer, mg_smoother, mg_smoother);
607 mg.set_edge_matrices(mg_interface, mg_interface);
608
609 PreconditionMG<spacedim,
610 LinearAlgebra::distributed::Vector<float>,
611 MGTransferMatrixFree<spacedim, float>>
612 preconditioner(dh, mg, mg_transfer);
613
614 auto invA = A;
615 invA = inverse_operator(A, cg_stiffness, preconditioner);
616#else
617 using Payload =
618 TrilinosWrappers::internal::LinearOperatorImplementation::TrilinosPayload;
619// LA::MPI::PreconditionAMG prec_A;
620// {
621// LA::MPI::PreconditionAMG::AdditionalData data;
622// # ifdef USE_PETSC_LA
623// data.symmetric_operator = true;
624// # endif
625// prec_A.initialize(stiffness_matrix, data);
626// }
627// const auto amgA = linear_operator<VectorType, VectorType>(A, prec_A);
628// invA = inverse_operator(A, cg_stiffness, amgA);
629#endif
630
631
632 // Some aliases
633 auto &u = solution.block(0);
634 auto &lambda = solution.block(1);
635
636 const auto &f = system_rhs.block(0);
637 const auto &g = system_rhs.block(1);
638
639 if (inclusions.n_dofs() == 0)
640 {
641 u = invA * f;
642 }
643 else
644 {
645#ifdef MATRIX_FREE_PATH
646 auto Bt =
647 linear_operator<VectorType, VectorType, Payload>(*coupling_operator);
648 Bt.reinit_range_vector = [this](VectorType &vec, const bool) {
649 vec.reinit(owned_dofs[0], relevant_dofs[0], mpi_communicator);
650 };
651 Bt.reinit_domain_vector = [this](VectorType &vec, const bool) {
652 vec.reinit(owned_dofs[1], relevant_dofs[1], mpi_communicator);
653 };
654
655 const auto B = transpose_operator<VectorType, VectorType, Payload>(Bt);
656#else
657 const auto Bt =
658 linear_operator<VectorType, VectorType, Payload>(coupling_matrix);
659 const auto B = transpose_operator<VectorType, VectorType, Payload>(Bt);
660#endif
661 const auto C =
662 linear_operator<VectorType, VectorType, Payload>(inclusion_matrix);
663
664 // Schur complement
665 const auto S = B * invA * Bt;
666
667 // Schur complement preconditioner
668 auto invS = S;
669 SolverCG<VectorType> cg_schur(par.outer_control);
670 invS = inverse_operator<Payload, SolverCG<VectorType>>(S, cg_schur);
671
672 pcout << " f norm: " << f.l2_norm() << ", g norm: " << g.l2_norm()
673 << std::endl;
674
675 // Compute Lambda first
676 lambda = invS * (B * invA * f - g);
677 pcout << " Solved for lambda in " << par.outer_control.last_step()
678 << " iterations." << std::endl;
679
680 // Then compute u
681 u = invA * (f - Bt * lambda);
682 pcout << " u norm: " << u.l2_norm()
683 << ", lambda norm: " << lambda.l2_norm() << std::endl;
684 }
685
686 pcout << " Solved for u in " << par.inner_control.last_step()
687 << " iterations." << std::endl;
688 constraints.distribute(u);
689 inclusion_constraints.distribute(lambda);
690 solution.update_ghost_values();
691 locally_relevant_solution = solution;
692}

697void
698PoissonProblem<dim, spacedim>::refine_and_transfer()
699{
700 TimerOutput::Scope t(computing_timer, "Refine");
701 Vector<float> error_per_cell(tria.n_active_cells());
702 KellyErrorEstimator<spacedim>::estimate(dh,
703 QGauss<spacedim - 1>(par.fe_degree +
704 1),
705 {},
706 locally_relevant_solution.block(0),
707 error_per_cell);
708 if (par.refinement_strategy == "fixed_fraction")
709 {
710 parallel::distributed::GridRefinement::refine_and_coarsen_fixed_fraction(
711 tria, error_per_cell, par.refinement_fraction, par.coarsening_fraction);
712 }
713 else if (par.refinement_strategy == "fixed_number")
714 {
715 parallel::distributed::GridRefinement::refine_and_coarsen_fixed_number(
716 tria,
717 error_per_cell,
718 par.refinement_fraction,
719 par.coarsening_fraction,
720 par.max_cells);
721 }
722 else if (par.refinement_strategy == "global")
723 for (const auto &cell : tria.active_cell_iterators())
724 cell->set_refine_flag();
725
726 parallel::distributed::SolutionTransfer<spacedim, VectorType> transfer(dh);
727 tria.prepare_coarsening_and_refinement();
728 inclusions.inclusions_as_particles.prepare_for_coarsening_and_refinement();
729 transfer.prepare_for_coarsening_and_refinement(
730 locally_relevant_solution.block(0));
731 tria.execute_coarsening_and_refinement();
732 inclusions.inclusions_as_particles.unpack_after_coarsening_and_refinement();
733 setup_dofs();
734 transfer.interpolate(solution.block(0));
735 constraints.distribute(solution.block(0));
736 locally_relevant_solution.block(0) = solution.block(0);
737}
